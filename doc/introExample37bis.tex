> ChebPolynomials = proc(n) {
    var i, res;
    if (n<0) then res = [||]
    else if (n==0) then res = [|1|]
    else {
       res = [|1, _x_|];
       for i from 2 to n do res[i] = horner(2*_x_*res[i-1]-res[i-2]);
    };
    return res;
  };
